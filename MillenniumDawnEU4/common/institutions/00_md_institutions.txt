# Millennium Dawn uses the eight native institution slots. Internal keys remain
# unchanged so engine and base-game scripted triggers stay compatible.

feudalism = {
	bonus = { trade_efficiency = 0.1 }
	trade_company_efficiency = 0.1
	history = {
		is_year = 2000
		is_city = yes
		owner = { always = yes }
		OR = { is_capital = yes has_port = yes development = 15 center_of_trade = 1 }
	}
	can_embrace = { always = yes }
	embracement_speed = {
		modifier = { factor = 0.6 scale = yes custom_trigger_tooltip = { tooltip = MD_INSTITUTION_NEARBY_GLOBALIZATION any_friendly_coast_border_province = { feudalism = 100 } } }
		modifier = { factor = 0.08 scale = yes custom_trigger_tooltip = { tooltip = MD_INSTITUTION_TRADE_PORT has_port = yes } }
		modifier = { factor = 0.05 scale = yes custom_trigger_tooltip = { tooltip = MD_INSTITUTION_CAPITAL is_capital = yes } }
		modifier = { factor = 1 scale = yes potential = { owner = { has_institution = feudalism } } custom_trigger_tooltip = { tooltip = MD_INSTITUTION_OWNER_EMBRACED owner = { has_institution = feudalism } } }
	}
	ai_will_do = { factor = 24 }
}
