# Millennium Dawn uses the eight native institution slots. Internal keys remain
# unchanged so engine and base-game scripted triggers stay compatible.

imperialism = {
	bonus = { diplomatic_reputation = 1 }
	historical_start_date = 1830.1.1
	history = { is_year = 2000 is_city = yes owner = { always = yes } }
	can_start = { is_year = 1830 is_city = yes is_capital = yes development = 15 }
	start_chance = 100
	can_embrace = { always = yes }
	embracement_speed = {
		modifier = { factor = 0.5 scale = yes any_friendly_coast_border_province = { imperialism = 100 } }
		modifier = { factor = 1 scale = yes potential = { owner = { has_institution = imperialism } } owner = { has_institution = imperialism } }
	}
	ai_will_do = { factor = 24 }
}
