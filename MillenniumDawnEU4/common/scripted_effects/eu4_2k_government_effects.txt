eu4_2k_normalize_modern_government = {
	if = {
		limit = {
			is_year = 1836
			government = monarchy
		}
		if = {
			limit = {
				OR = {
					has_reform = parliamentary_reform
					has_reform = multi_party_system
				}
			}
			add_government_reform = eu4_2k_parliamentary_monarchy_reform
		}
		else = {
			add_government_reform = eu4_2k_absolute_monarchy_reform
		}
	}

	if = {
		limit = {
			is_year = 1836
			government = republic
		}
		if = {
			limit = { has_reform = communist_government }
			add_government_reform = eu4_2k_socialist_state_reform
		}
		else_if = {
			limit = {
				OR = {
					has_reform = authoritarian_rule
					has_reform = dictatorial_rule
					has_reform = single_party_system
					has_reform = military_government
					has_reform = fascist_government
					has_reform = presidential_despot_reform
				}
			}
			add_government_reform = eu4_2k_authoritarian_republic_reform
		}
		else_if = {
			limit = { has_reform = parliamentary_reform }
			add_government_reform = eu4_2k_parliamentary_republic_reform
		}
		else = {
			add_government_reform = eu4_2k_presidential_republic_reform
		}
	}
}
